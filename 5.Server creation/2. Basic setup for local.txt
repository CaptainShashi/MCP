# Install uv

# Create a project folder named fastmcp-demo-server
mkdir fastmcp-demo-server
cd fastmcp-demo-server

# Open the folder in VS Code
code .

# Open terminal

# Execute the command
uv init .

# Add fastmcp
uv add fastmcp

# Fastmcp version
uv run python -c "import fastmcp; print(fastmcp.__version__)"

# Create a basic server
# (create main.py file manually)

# Test the server
uv run fastmcp dev main.py          ( opens a page same like postman )

# Run the server
uv run fastmcp run main.py

# Add the server to Claude Desktop
uv run fastmcp install claude-desktop main.py
